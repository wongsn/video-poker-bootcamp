<button id="showRightPush">Show/Hide Right Push Menu</button>

const menuRight = document.getElementById( 'cbp-spmenu-s2' ); //the menu

const showRightPush = document.getElementById( 'showRightPush' );

showRightPush.onclick = function() {
	classList.toggle( this, 'active' );
	classList.toggle( body, 'cbp-spmenu-push-toleft' );
	classList.toggle( menuRight, 'cbp-spmenu-open' );
	disableOther( 'showRightPush' );
};



.cbp-spmenu-push-toleft {
	left: -240px;
}
.cbp-spmenu-left.cbp-spmenu-open {
	left: 0px;
}

.cbp-spmenu-right.cbp-spmenu-open {
	right: 0px;
}
